```python Python
import asyncio
import logging
from meshagent.api import RoomClient
from meshagent.otel import otel_config
from meshagent.agents.llmrunner import LLMTaskRunner
from meshagent.openai import OpenAIResponsesAdapter
from meshagent.api.messaging import JsonContent, TextContent

otel_config(service_name="taskrunner")
log = logging.getLogger("taskrunner")


async def run_once(room_name: str, prompt: str):
    """
    Connect to a MeshAgent room and execute a TaskRunner directly (no toolkit registration).

    Args:
        room_name: Target room (will be created on first connect if it doesn't exist)
        prompt: The text to send to the task runner
    """
    # Run with:
    # meshagent room connect --room=quickstart --identity=sample-participant -- python3 run_taskrunner.py
    try:
        async with RoomClient() as room:
            log.info(f"Connected to room: {room.room_name}")
            runner = LLMTaskRunner(
                llm_adapter=OpenAIResponsesAdapter(),
                supports_tools=True,
            )
            response = await runner.run(
                room=room,
                arguments={"prompt": prompt, "tools": [], "model": None},
            )
            if isinstance(response, JsonContent):
                log.info(response.json)
            elif isinstance(response, TextContent):
                log.info(response.text)
            else:
                log.info(response)
            return response
    except Exception as e:
        log.error(f"Connection failed:{e}")
        raise


asyncio.run(
    run_once(
        room_name="quickstart",
        prompt="Write a product description for a bluetooth speaker",
    )
)
```
