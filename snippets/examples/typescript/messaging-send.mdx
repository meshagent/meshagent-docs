```typescript TypeScript
import { RoomClient } from "@meshagent/meshagent";

// Run this in one terminal, then join a second participant in the same room:
// meshagent room connect --room=example-room --identity=alice -- <your node command>

async function main(): Promise<void> {
  const room = new RoomClient();

  try {
    await room.start();
    room.messaging.enable();
    room.messaging.on("message", (event) => {
      console.log(
        `received a message from ${event.message.fromParticipantId} with type '${event.message.type}':`,
        event.message.message,
      );
    });

    console.log("Messaging enabled. Waiting for remote participants...");
    await new Promise((resolve) => setTimeout(resolve, 1000));

    const remoteParticipant = room.messaging.remoteParticipants[0];
    if (remoteParticipant == null) {
      console.log("No remote participants are connected yet.");
      await new Promise((resolve) => setTimeout(resolve, 5000));
      return;
    }

    await room.messaging.sendMessage({
      to: remoteParticipant,
      type: "hello",
      message: {
        text: `Hello from ${room.localParticipant?.id}`,
      },
    });
    console.log(`Sent a message to ${remoteParticipant.id}.`);
    await new Promise((resolve) => setTimeout(resolve, 3000));
  } finally {
    room.dispose();
  }
}

void main();
```
