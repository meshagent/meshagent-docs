```yaml Yaml
kind: Service
version: v1
metadata:
  name: voice1
  description: "A voice powered agent that you can talk to."
  annotations:
    meshagent.service.id: meshagent.voicebot1
agents:                                        
  - name: voice1
    annotations:
      meshagent.agent.type: "VoiceBot"
container:
  image: "meshagent/cli:latest"
  command: "/usr/bin/meshagent voicebot join --agent-name=voice1 --room-rules agents/voice1/rules.md"
  environment:
    - name: MESHAGENT_TOKEN
      token:
        identity: voice1
        role: agent

```
