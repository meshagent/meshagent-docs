```yaml Yaml
version: v1
kind: Service
metadata:
  name: mytaskrunner
  annotations:
    meshagent.service.id: mytaskrunner
agents:
- name: mytaskrunner
  annotations:
    meshagent.agent.type: TaskRunner

    --room-rules agents/mytaskrunner/rules.txt --rule 'You are a helpful assistant'
  environment:
  - name: MESHAGENT_TOKEN
    token:
      identity: mytaskrunner
```
