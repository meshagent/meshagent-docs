```python Python
#!/usr/bin/env python3

"""
Scan for meshagent.yaml files and print shell exports with their JSON content.
"""

import argparse
import asyncio
import json
import pathlib
import sys

import aiofiles
import aiofiles.ospath
import yaml


parser = argparse.ArgumentParser(description="Process meshagent manifests.")
parser.add_argument(
    "--folder",
    type=pathlib.Path,
    required=True,
    default=pathlib.Path(__file__).resolve().parents[1],
    help="Root folder containing meshagent.yaml files",
)
args = parser.parse_args()
ROOT = args.folder.resolve()


async def process_manifest(yml_path: pathlib.Path) -> str | None:
    has_manifest = await aiofiles.ospath.exists(yml_path)
    if not has_manifest:
        return None

    try:
        async with aiofiles.open(yml_path, "r") as file:
            raw_manifest = await file.read()
            manifest = yaml.safe_load(raw_manifest)

            json_content = json.dumps(manifest)
            parts = yml_path.relative_to(ROOT).parts[:-1]
            var = "LABEL_" + "_".join(parts).upper()
            escaped_json = json_content.replace('"', '\\"')

            return f'export {var}="{escaped_json}"'
    except Exception as exc:
        print(f"Error processing {yml_path}: {exc}", file=sys.stderr)
        return None


async def main() -> None:
    tasks = [process_manifest(yml) for yml in ROOT.rglob("meshagent.yaml")]
    results = await asyncio.gather(*tasks, return_exceptions=True)

    for result in results:
        if isinstance(result, str):
            print(result)
        elif isinstance(result, Exception):
            print(f"Error: {result}", file=sys.stderr)


if __name__ == "__main__":
    asyncio.run(main())
```
