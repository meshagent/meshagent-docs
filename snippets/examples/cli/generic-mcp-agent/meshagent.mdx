```yaml Yaml
kind: Service
version: v1
metadata:
  name: mcp-assistant
  description: "A process-backed agent that can use MCP tools in the room"
  annotations:
    meshagent.service.id: "mcp-assistant"
agents:
  - name: mcp-assistant
    description: "Research assistant with access to MCP tools"
    annotations:
      meshagent.agent.type: "ChatBot"
container:
  image: "meshagent/cli:default"
  command: >-
    meshagent process join
    --agent-name mcp-assistant
    --channel chat
    --require-web-search
    --mcp
    --rule "You are a research assistant. Use the available MCP tools when users ask about them."
    --room-rules "agents/mcp-assistant/rules.txt"
  environment:
    - name: MESHAGENT_TOKEN
      token:
        identity: mcp-assistant
```
