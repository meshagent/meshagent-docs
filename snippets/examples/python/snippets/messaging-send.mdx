```python Python
import asyncio
from meshagent.api import RoomClient


async def main():
    # Run this in one terminal, then join a second participant in the same room:
    # meshagent room connect --room=example-room --identity=alice -- python3 messaging-send.py
    async with RoomClient() as room:
        room.messaging.enable()

        def on_message(*, message):
            print(
                f"received a message from {message.from_participant_id} with type '{message.type}': {message.message}"
            )

        room.messaging.on("message", on_message)
        print("Messaging enabled. Waiting for remote participants...")

        await asyncio.sleep(1.0)
        remote_participant = next(iter(room.messaging.remote_participants), None)
        if remote_participant is None:
            print("No remote participants are connected yet.")
            await asyncio.sleep(5.0)
            return

        await room.messaging.send_message(
            to=remote_participant,
            type="hello",
            message={"text": f"Hello from {room.local_participant.id}"},
        )
        print(f"Sent a message to {remote_participant.id}.")
        await asyncio.sleep(3.0)


asyncio.run(main())
```
