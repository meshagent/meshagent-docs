```yaml Yaml
kind: ServiceTemplate
version: v1
metadata:
  name: support-suite
  description: "Chatbot + mailbot template for support teams"
agents:
  - name: support-chatbot
    description: "Chatbot for live support questions"
    annotations:
      meshagent.agent.type: "ChatBot"
  - name: support-mailbot
    description: "Mailbot that handles support inbox"
    annotations:
      meshagent.agent.type: "MailBot"
variables:
  - name: support_email
    type: email
    description: "Email address for the mailbot (e.g. support@company.com)"
ports:
  - num: "*"
    type: http
    liveness: "/"
    endpoints:
      - path: /chat
        meshagent:
          identity: support-chatbot
      - path: /mail
        meshagent:
          identity: support-mailbot
container:
  image: "us-central1-docker.pkg.dev/meshagent-public/images/cli:{SERVER_VERSION}-esgz"
  command:
    /usr/bin/meshagent multi service -c "
    chatbot --agent-name=support-chatbot --path=/chat --require-web-search
    --rule='You are a helpful support chatbot.';
    mailbot --agent-name=support-mailbot --path=/mail --queue=support_inbox
    --email-address={{ support_email }} --toolkit-name=email --require-web-search
    --rule='You handle incoming support emails.'"
  storage:
    room:
      - path: /data
        read_only: false

```
