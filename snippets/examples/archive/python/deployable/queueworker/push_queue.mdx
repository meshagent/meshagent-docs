```python Python
import asyncio
import logging
from meshagent.api import RoomClient
from meshagent.otel import otel_config

otel_config(service_name="worker")
log = logging.getLogger("worker")


async def push():
    # Run with:
    # meshagent room connect --room=quickstart --identity=sample-participant -- python3 push_queue.py
    try:
        async with RoomClient() as room:
            log.info(f"Connected to room: {room.room_name}")
            await room.queues.send(
                name="storage-worker-queue",
                message={"instructions": "save a poem about ai to poem.txt"},
            )
    except Exception as e:
        log.error(f"Connection failed:{e}")
        raise


asyncio.run(push())
```
