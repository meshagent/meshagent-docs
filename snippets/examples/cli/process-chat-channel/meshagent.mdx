```yaml Yaml
kind: Service
version: v1
metadata:
  name: support-agent-chat
  description: "A process-backed support agent with a chat channel"
  annotations:
    meshagent.service.id: "support-agent-chat"
agents:
  - name: support-agent
    description: "A chat-first process-backed support agent"
    annotations:
      meshagent.agent.type: "ChatBot"
      meshagent.chatbot.threading: "default-new"
      meshagent.chatbot.thread-list: ".threads/support-agent/index.threadl"
container:
  image: "meshagent/cli:default"
  command: >-
    /usr/bin/meshagent process join
    --agent-name=support-agent
    --channel=chat
    --threading-mode=default-new
    --thread-dir=".threads/support-agent"
    --require-web-search
    --require-storage
    --room-rules="agents/support-agent/rules.md"
    --rule="You are a helpful support agent. Answer clearly, use web search when needed, and save important artifacts to storage."
  environment:
    - name: MESHAGENT_TOKEN
      token:
        identity: support-agent
  storage:
    room:
      - path: /data
        read_only: false

```
