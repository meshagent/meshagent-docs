```python Python
import asyncio

from meshagent.api.client import Meshagent
from meshagent.cli.auth_async import get_access_token
from meshagent.cli.local_settings import resolve_api_url

async def main():
    access_token = await get_access_token()
    if access_token is None:
        raise RuntimeError('Run "meshagent auth login" first.')

    client = Meshagent(base_url=resolve_api_url(), token=access_token)
    try:
        projects = await client.list_projects()
        print(projects)
    finally:
        await client.close()

asyncio.run(main())
```
