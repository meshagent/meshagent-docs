```yaml Yaml
kind: Service
version: v1
metadata:
  name: basic-chatbot
  description: "Expose a ChatBot with built in MeshAgent tools"
agents:
  - name: basic-chatbot
    description: "Conversational agent for the room"
    annotations:
      meshagent.agent.type: "ChatBot"
ports:
- num: "*" 
  type: http
  liveness: "/"
  endpoints:
  - path: /chat
    meshagent:
      identity: "basic-chatbot"
      # api: if you need to use secrets/OAuth with the agent you can add those permissions here
container:
  image: "IMAGE:TAG"
```
